(defn gh-get-text [z-xml attr]
  (cond
    (keyword? attr)
    (zx/xml-> z-xml m-k attr zx/text)
    (= 2 (count attr))
    (zx/xml-> z-xml m-k (first attr) (second attr) zx/text)
    (= 3 (count attr))
    (zx/xml-> z-xml m-k (first attr) (second attr)
              (nth attr 2) zx/text)))
